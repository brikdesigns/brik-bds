# Pricing card

A pricing tier display card with title, price, features list, and call-to-action. Supports a highlighted state for the recommended plan.

## Usage

```tsx
import { PricingCard } from '@brik/bds';
import { Badge } from '@brik/bds';
import { Button } from '@brik/bds';

// Standard pricing card
<PricingCard
  title="Professional"
  price="$49"
  period="/month"
  description="For growing businesses."
  features={['Unlimited projects', 'Priority support']}
  action={<Button variant="primary" fullWidth>Get started</Button>}
/>

// Highlighted (recommended) plan
<PricingCard
  title="Professional"
  price="$49"
  period="/month"
  badge={<Badge>Most popular</Badge>}
  features={['Unlimited projects', 'Priority support']}
  action={<Button variant="primary" fullWidth>Get started</Button>}
  highlighted
/>
```
