## Usage

```tsx
import { Select } from '@brik/bds';

const options = [
  { label: 'First choice', value: 'first' },
  { label: 'Second choice', value: 'second' },
  { label: 'Third choice', value: 'third' },
];

<Select
  placeholder="Select one..."
  options={options}
  defaultValue="second"
/>
```
