## Usage

```tsx
import { Dot } from '@brik/bds';

// Basic status indicator
<Dot status="positive" />

// Different sizes
<Dot size="sm" status="warning" />
<Dot size="md" status="error" />
<Dot size="lg" status="info" />

// With label
<div style={{ display: 'flex', alignItems: 'center', gap: '8px' }}>
  <Dot status="positive" />
  <span>Online</span>
</div>
```
