## Usage

```tsx
import { Dialog } from '@brik/bds';
import { useState } from 'react';

const [isOpen, setIsOpen] = useState(false);

<Dialog
  isOpen={isOpen}
  onClose={() => setIsOpen(false)}
  title="Save changes?"
  description="You have unsaved changes that will be lost."
  confirmLabel="Save"
  onConfirm={handleSave}
/>
```
