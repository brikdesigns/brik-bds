## Usage

```tsx
import { Switch } from '@brik/bds';

// Uncontrolled
<Switch label="Enable feature" size="lg" defaultChecked />

// Controlled
<Switch
  label="Dark mode"
  checked={darkMode}
  onChange={(e) => setDarkMode(e.target.checked)}
/>
```
