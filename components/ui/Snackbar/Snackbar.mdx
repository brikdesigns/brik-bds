# Snackbar

Brief notification bar that appears at the edge of the screen. Use for non-critical feedback that doesn't require user action, like "Item saved" or "Connection restored."

## Usage

```tsx
import { Snackbar } from '@brik/bds';
import { useState } from 'react';

const [isOpen, setIsOpen] = useState(false);

// Basic snackbar
<Snackbar
  isOpen={isOpen}
  onClose={() => setIsOpen(false)}
  message="Changes saved"
/>

// With action and variant
<Snackbar
  isOpen={isOpen}
  onClose={() => setIsOpen(false)}
  message="Item deleted"
  variant="error"
  actionLabel="Undo"
  onAction={handleUndo}
/>

// Custom position
<Snackbar
  isOpen={isOpen}
  onClose={() => setIsOpen(false)}
  message="Connected"
  variant="success"
  position="top"
/>
```
