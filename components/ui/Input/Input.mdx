## Usage

```tsx
import { Input } from '@brik/bds';

// Basic input
<Input placeholder="Enter text..." />

// With label
<Input label="Email Address" placeholder="you@example.com" type="email" />

// With helper text
<Input
  label="Password"
  type="password"
  helperText="Must be at least 8 characters"
/>

// Error state
<Input
  label="Email"
  error="Please enter a valid email address"
  type="email"
/>

// Sizes
<Input size="sm" label="Small" />
<Input size="md" label="Medium" />
<Input size="lg" label="Large" />

// Full width
<Input fullWidth label="Full Name" />
```
