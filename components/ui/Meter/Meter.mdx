## Usage

```tsx
import { Meter } from '@brikdesigns/bds';

// Basic meter
<Meter value={82} max={100} status="positive" label="Score" />

// With custom formatter
<Meter
  value={0.73}
  max={1}
  status="warning"
  label="Completion"
  valueFormatter={(v, max) => `${Math.round((v / max) * 100)}%`}
/>

// Small size, label below
<Meter value={45} max={100} size="sm" labelPosition="below" />
```
