## Usage

```tsx
import { CardDisplay, Badge, Button } from '@brik/bds';

<CardDisplay
  imageSrc="/product.jpg"
  imageAlt="Product photo"
  badge={<Badge size="sm">New</Badge>}
  title="Product name"
  description="Short product description."
  action={<Button variant="primary" size="sm">View details</Button>}
/>
```
