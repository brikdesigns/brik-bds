## Usage

```tsx
import { Slider } from '@brik/bds';
import { useState } from 'react';

const [value, setValue] = useState(50);

// Basic slider
<Slider label="Volume" value={value} onChange={setValue} showValue />

// Custom range with step
<Slider
  label="Price"
  value={value}
  onChange={setValue}
  min={0}
  max={1000}
  step={50}
  size="lg"
  showValue
/>
```
