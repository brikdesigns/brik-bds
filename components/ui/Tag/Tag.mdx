# Tag

Categorization and filtering labels with optional left/right icons and remove functionality. Tags help organize content and provide visual context for filtering and selection. Available in three sizes (sm, md, lg).

## Usage

```tsx
import { Tag } from '@brik/bds';
import { FontAwesomeIcon } from '@fortawesome/react-fontawesome';
import { faCertificate, faCircleXmark } from '@fortawesome/free-solid-svg-icons';

// Basic tag
<Tag>Category</Tag>

// Sizes
<Tag size="sm">Small</Tag>
<Tag size="md">Medium</Tag>
<Tag size="lg">Large</Tag>

// Left icon
<Tag icon={<FontAwesomeIcon icon={faCertificate} />}>Featured</Tag>

// Right icon
<Tag trailingIcon={<FontAwesomeIcon icon={faCircleXmark} />}>Dismiss</Tag>

// Both icons
<Tag icon={<FontAwesomeIcon icon={faCertificate} />} trailingIcon={<FontAwesomeIcon icon={faCircleXmark} />}>
  Both
</Tag>

// Removable tag
<Tag onRemove={() => console.log('Removed')}>Filter</Tag>

// Disabled tag
<Tag disabled>Archived</Tag>
```
