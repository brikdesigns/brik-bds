## Usage

```tsx
import { Stepper } from '@brik/bds';
import { useState } from 'react';

const [count, setCount] = useState(1);

// Basic stepper
<Stepper value={count} onChange={setCount} />

// With bounds and custom step
<Stepper
  value={count}
  onChange={setCount}
  min={0}
  max={100}
  step={5}
  size="lg"
/>
```
