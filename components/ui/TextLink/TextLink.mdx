## Usage

```tsx
import { TextLink } from '@brikdesigns/bds';

// Default link
<TextLink href="/about">Learn more</TextLink>

// Small link
<TextLink href="/details" size="small">View details</TextLink>

// External link
<TextLink
  href="https://example.com"
  target="_blank"
  rel="noopener noreferrer"
>
  Visit website
</TextLink>
```
