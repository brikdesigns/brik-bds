## Usage

```tsx
import { TextLink } from '@brik/bds';

// Default link
<TextLink href="/about">Learn More</TextLink>

// Small link
<TextLink href="/details" size="small">View Details</TextLink>

// External link
<TextLink
  href="https://example.com"
  target="_blank"
  rel="noopener noreferrer"
>
  Visit Website
</TextLink>
```
