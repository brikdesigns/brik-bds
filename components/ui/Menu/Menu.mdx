# Menu

Floating dropdown panel for navigation and action menus. Contains selectable items with optional icons. Closes on outside click or Escape key.

## Usage

```tsx
import { Menu } from '@brik/bds';
import { Button } from '@brik/bds';

const [open, setOpen] = useState(false);

<div style={{ position: 'relative' }}>
  <Button onClick={() => setOpen(!open)}>Options</Button>
  <Menu
    isOpen={open}
    onClose={() => setOpen(false)}
    items={[
      { id: '1', label: 'Edit', icon: <EditIcon />, onClick: handleEdit },
      { id: '2', label: 'Delete', onClick: handleDelete },
    ]}
    style={{ top: '100%', left: 0 }}
  />
</div>
```
