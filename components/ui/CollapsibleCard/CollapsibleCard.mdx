## Usage

```tsx
import { CollapsibleCard } from '@brik/bds';

// Basic collapsible card
<CollapsibleCard
  sectionLabel="Section 01"
  title="Overview and Goals"
>
  <p>Content goes here...</p>
</CollapsibleCard>

// Initially open
<CollapsibleCard
  title="Settings"
  defaultOpen
>
  <p>Settings content...</p>
</CollapsibleCard>
```
