## Usage

```tsx
import { Chip } from '@brik/bds';

// Basic chip
<Chip label="Design" />

// Removable chip
<Chip label="Filter" onRemove={() => removeFilter('design')} />

// Size and variant
<Chip label="Active" size="sm" variant="primary" appearance="light" />

// Disabled
<Chip label="Archived" disabled />
```
