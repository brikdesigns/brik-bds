## Usage

```tsx
import { Popover } from '@brik/bds';

// Click-triggered popover
<Popover content={<p>Additional details here</p>}>
  <button>More info</button>
</Popover>

// Hover-triggered, top placement
<Popover
  content={<p>Tooltip-style content</p>}
  trigger="hover"
  placement="top"
>
  <span>Hover me</span>
</Popover>
```
