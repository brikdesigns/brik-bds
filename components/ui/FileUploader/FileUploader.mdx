## Usage

```tsx
import { FileUploader } from '@brik/bds';

// Basic file uploader
<FileUploader
  accept=".pdf,.jpg,.png"
  helperText="PDF, JPG, or PNG up to 10MB"
  onChange={(files) => handleFiles(files)}
/>

// With size limit and multiple files
<FileUploader
  accept="image/*"
  multiple
  maxSize={5 * 1024 * 1024}
  label="Upload images"
  helperText="Select one or more images"
  onChange={(files) => handleFiles(files)}
/>
```
