## Usage

```tsx
import { FileUploader } from '@brik/bds';

// Basic file uploader
<FileUploader
  accept=".pdf,.jpg,.png"
  onChange={(files) => handleFiles(files)}
/>

// With size limit and multiple files
<FileUploader
  accept="image/*"
  multiple
  maxSize={5 * 1024 * 1024}
  onChange={(files) => handleFiles(files)}
/>
```
