Dialog overlay component with multiple size variants and configurable close behavior. Use for focused tasks, confirmations, and forms that require user attention.

## Usage

```tsx
import { Modal } from '@brik/bds';
import { Button } from '@brik/bds';
import { useState } from 'react';

const [isOpen, setIsOpen] = useState(false);

<>
  <Button onClick={() => setIsOpen(true)}>Open Modal</Button>

  <Modal
    isOpen={isOpen}
    onClose={() => setIsOpen(false)}
    title="Modal Title"
    size="md"
    closeOnBackdrop={true}
    closeOnEscape={true}
    showCloseButton={true}
    footer={
      <div style={{ display: 'flex', gap: '16px' }}>
        <Button onClick={() => setIsOpen(false)}>Cancel</Button>
        <Button onClick={() => { /* action */ setIsOpen(false); }}>
          Confirm
        </Button>
      </div>
    }
  >
    <p>Modal content goes here.</p>
  </Modal>
</>
```
